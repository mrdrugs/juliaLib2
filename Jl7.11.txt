function factor(n)
    divs=[]
    div_powers=[]
    spqr=convert(Int,(ceil(sqrt(n))))
    i=2
    original=n
    while i*i<=original
        if n%i==0
            if length(divs)==0 || divs[lastindex(divs)]!=i
                push!(divs,i)
                push!(div_powers,1)
            else
                div_powers[lastindex(div_powers)]+=1
            end
            n=nÃ·i
        else
            i+=1
        end
    end
    if n!=1
        if divs[lastindex(divs)]==n
            div_powers[lastindex(div_powers)]+=1
        else
            push!(divs,n)
            push!(div_powers,1)
        end
    end
    return divs, div_powers
end

function nilpotents(n)
    divs, _ = factor(n)
    multip=1
    for x in divs
        multip*=x
    end
    return multip:multip:(n-1)
end
